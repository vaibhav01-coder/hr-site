Run these SQL commands in the Supabase SQL Editor:

4.1 candidates table
create table candidates (
  id          uuid default gen_random_uuid() primary key,
  full_name   text not null,
  mobile      text not null,
  whatsapp    text,
  age         integer,
  city        text,
  qualification text,
  trade       text,
  company_pref  text,
  shift_pref    text,
  aadhar_last4  text,
  resume_url    text,
  created_at    timestamp default now()
);

4.2 jobs table
create table jobs (
  id                uuid default gen_random_uuid() primary key,
  company_name      text not null,
  education_required text,
  salary            text,
  shift             text,
  perks             text,
  vacancies         integer,
  is_active         boolean default true,
  created_at        timestamp default now()
);

4.3 applications table
create table applications (
  id                  uuid default gen_random_uuid() primary key,
  candidate_id        uuid references candidates(id),
  job_id              uuid references jobs(id),
  status              text default 'under_review',
  payment_intent_id   text,
  payment_status      text default 'pending',
  refund_id           text,
  refund_status       text default 'none',
  created_at          timestamp default now()
);

5.4 routes/jobs.js
const router   = require('express').Router();
const supabase = require('../config/supabase');

// GET all active jobs
router.get('/', async (req, res) => {
  const { data, error } = await supabase
    .from('jobs').select('*').eq('is_active', true);
  if (error) return res.status(500).json({ error });
  res.json(data);
});

// GET single job
router.get('/:id', async (req, res) => {
  const { data, error } = await supabase
    .from('jobs').select('*').eq('id', req.params.id).single();
  if (error) return res.status(404).json({ error });
  res.json(data);
});

// POST create job (HR)
router.post('/', async (req, res) => {
  const { data, error } = await supabase
    .from('jobs').insert([req.body]).select();
  if (error) return res.status(500).json({ error });
  res.status(201).json(data[0]);
});

module.exports = router;

5.6 routes/payment.js
const router   = require('express').Router();
const stripe   = require('../config/stripe');
const supabase = require('../config/supabase');

// POST create Stripe payment intent — ₹50
router.post('/create-intent', async (req, res) => {
  const { candidate_id } = req.body;
  const intent = await stripe.paymentIntents.create({
    amount:   5000,       // ₹50 in paise
    currency: 'inr',
    metadata: { candidate_id }
  });
  res.json({ clientSecret: intent.client_secret,
             intentId:     intent.id });
});

// POST verify payment
router.post('/verify', async (req, res) => {
  const { payment_intent_id } = req.body;
  const intent = await stripe.paymentIntents
                              .retrieve(payment_intent_id);
  if (intent.status === 'succeeded')
    return res.json({ success: true, status: 'paid' });
  res.status(400).json({ success: false,
                         status: intent.status });
});

// POST refund — HR rejects candidate
router.post('/refund/:application_id', async (req, res) => {
  const { data: app } = await supabase
    .from('applications').select('*')
    .eq('id', req.params.application_id).single();

  const refund = await stripe.refunds.create({
    payment_intent: app.payment_intent_id,
    amount: 5000
  });

  await supabase.from('applications').update({
    refund_id:     refund.id,
    refund_status: 'refunded',
    status:        'rejected'
  }).eq('id', req.params.application_id);

  res.json({ success: true, refund_id: refund.id });
});

module.exports = router;

5.7 routes/applications.js
const router   = require('express').Router();
const stripe   = require('../config/stripe');
const supabase = require('../config/supabase');

// POST submit application (after payment)
router.post('/', async (req, res) => {
  const { candidate_id, job_id, payment_intent_id } = req.body;
  const intent = await stripe.paymentIntents
                              .retrieve(payment_intent_id);
  if (intent.status !== 'succeeded')
    return res.status(400).json({ error: 'Payment not done' });

  const { data, error } = await supabase
    .from('applications')
    .insert([{ candidate_id, job_id,
               payment_intent_id,
               payment_status: 'paid',
               status: 'under_review' }])
    .select();

  if (error) return res.status(500).json({ error });
  res.status(201).json({ success: true, application: data[0] });
});

// GET all applications (HR view)
router.get('/', async (req, res) => {
  const { data, error } = await supabase
    .from('applications')
    .select('*, candidates(*), jobs(*)');
  if (error) return res.status(500).json({ error });
  res.json(data);
});

// PATCH update application status (HR)
router.patch('/:id/status', async (req, res) => {
  const { status } = req.body;
  const { data, error } = await supabase
    .from('applications')
    .update({ status })
    .eq('id', req.params.id).select();
  if (error) return res.status(500).json({ error });
  res.json({ success: true, application: data[0] });
});

module.exports = router;

5.8 routes/upload.js — Resume to Supabase Storage
const router   = require('express').Router();
const multer   = require('multer');
const supabase = require('../config/supabase');

const upload = multer({ storage: multer.memoryStorage() });

router.post('/resume', upload.single('resume'),
  async (req, res) => {
    const file     = req.file;
    const fileName = `resumes/${Date.now()}_${file.originalname}`;

    const { error } = await supabase.storage
      .from('resumes')
      .upload(fileName, file.buffer,
              { contentType: 'application/pdf' });

    if (error) return res.status(500).json({ error });

    const { data: urlData } = supabase.storage
      .from('resumes').getPublicUrl(fileName);

    res.json({ success: true, url: urlData.publicUrl });
  }
);

module.exports = router;
